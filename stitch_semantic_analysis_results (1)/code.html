import { OpenAIEmbeddings } from 'langchain/embeddings/openai';
// Initialize the comparison engine
const embeddings = new OpenAIEmbeddings();
async function analyzeUserCompatibility(userA, userB) {
  try {
    // Fetch high-dimensional vectors
    const [vecA, vecB] = await Promise.all([
      embeddings.embedQuery(userA.text),
      embeddings.embedQuery(userB.text)
    ]);
    // Compute cosine similarity
    const similarity = cosineSimilarity(vecA, vecB);
    // Generate summary with LLM
    const summary = await llm.predict(
      `Explain compatibility score ${similarity} for users...`
    );
    return { score: similarity, summary };
  } catch (err) {
    console.error("Analysis failed", err);
  }
}
